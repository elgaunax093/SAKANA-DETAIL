    <script>
        gsap.registerPlugin(ScrollTrigger);

        const canvas = document.getElementById("animation-canvas");
        const ctx = canvas.getContext("2d");
        const loader = document.getElementById("loader");
        const loaderText = document.getElementById("loader-percentage");

        const frameCount = 100;
        const currentFramePath = index => `MULTIMEDIA/FRAMES/frame_${index.toString().padStart(4, '0')}.jpg`;
        
        const images = [];
        const canvasTimeline = { frame: 0 };
        let loadedFramesCount = 0;

        // 1. Precarga Inteligente Inmune a Bloqueos
        function startPreloading() {
            for (let i = 0; i < frameCount; i++) {
                const img = new Image();
                img.src = currentFramePath(i);
                img.onload = onFrameLoadSuccess;
                img.onerror = onFrameLoadError;
                images.push(img);
            }
        }

        function onFrameLoadSuccess() {
            loadedFramesCount++;
            refreshLoaderProgress();
        }

        function onFrameLoadError(e) {
            loadedFramesCount++;
            console.warn("Fotograma omitido estratégicamente para evitar freeze del preloader.");
            refreshLoaderProgress();
        }

        function refreshLoaderProgress() {
            const pct = Math.round((loadedFramesCount / frameCount) * 100);
            loaderText.innerText = pct + "%";

            if (loadedFramesCount >= frameCount) {
                launchAppUI();
            }
        }

        // Sistema de tiempo límite redundante (3.5s Máximo)
        setTimeout(() => {
            if (loader.style.display !== 'none') {
                console.log("Preloader watchdog activado.");
                launchAppUI();
            }
        }, 3500);

        function launchAppUI() {
            gsap.to(loader, {
                opacity: 0,
                duration: 0.6,
                onComplete: () => {
                    loader.style.display = "none";
                    buildScrollPipeline();
                }
            });
        }

        // 2. Escalado Adaptativo de Imagen (Cover Profesional)
        function resizeAppCanvas() {
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
            drawFrame();
        }

        function drawFrame() {
            const activeImg = images[canvasTimeline.frame];
            if (activeImg) {
                ctx.clearRect(0, 0, canvas.width, canvas.height);
                const wRatio = canvas.width / activeImg.width;
                const hRatio = canvas.height / activeImg.height;
                const bestRatio = Math.max(wRatio, hRatio);
                
                const offsetX = (canvas.width - activeImg.width * bestRatio) / 2;
                const offsetY = (canvas.height - activeImg.height * bestRatio) / 2;
                
                ctx.drawImage(activeImg, 0, 0, activeImg.width, activeImg.height,
                              offsetX, offsetY, activeImg.width * bestRatio, activeImg.height * bestRatio);
            }
        }

        // 3. Pipeline de Scroll por Inercia (Sin Atascos y Bidireccional)
        function buildScrollPipeline() {
            resizeAppCanvas();
            window.addEventListener("resize", resizeAppCanvas);

            // Una ráfaga fluida controlada por el avance físico del scroll
            gsap.to(canvasTimeline, {
                frame: frameCount - 1,
                snap: "frame",
                ease: "none",
                scrollTrigger: {
                    trigger: "#hero",
                    start: "top top",
                    end: "bottom top",
                    scrub: 1.2, // Factor LERP por inercia para máxima sedosidad de reproducción
                    pin: true,
                    onUpdate: drawFrame
                }
            });

            // Forzar renderizado del cuadro base inicial
            drawFrame();
        }

        // Inicialización nativa directa
        startPreloading();
    </script>
